// ── Cricbuzz API Scoring Integration ── window.syncDailyScores = async function(matchIdInput) { const apiKey = "6d53928bfdmsh545332aded830a3p11bdaajsncf079fc57095"; // Your verified key const matchId = matchIdInput || prompt("Enter Match ID (e.g., 104367):", "104367"); if(!matchId) return; window.showAlert("⚡ Connecting to Cricbuzz API...", "info"); try { const response = await fetch(`https://cricbuzz-cricket.p.rapidapi.com/mcenter/v1/${matchId}/hscard`, { method: 'GET', headers: { 'x-rapidapi-key': apiKey, 'x-rapidapi-host': 'cricbuzz-cricket.p.rapidapi.com', 'Content-Type': 'application/json' } }); if (!response.ok) throw new Error(`API Error: ${response.status}`); const data = await response.json(); if (!data.scoreCard) throw new Error("No scorecard data found for this ID."); // 1. Auto-fill Match Meta document.getElementById('matchLabel').value = data.matchHeader?.matchDescription || `Match ${matchId}`; document.getElementById('mfWinner').value = (data.matchHeader?.result?.winningTeam || "").toUpperCase(); document.getElementById('mfMotm').value = data.matchHeader?.playersOfTheMatch?.[0]?.name || ""; // 2. Clear existing rows to prevent duplicates document.getElementById('battingRows').innerHTML = ''; document.getElementById('bowlingRows').innerHTML = ''; battingRowCount = 0; bowlingRowCount = 0; // 3. Process Innings Data data.scoreCard.forEach(innings => { // --- Batting Sync --- const batsmen = innings.batTeamDetails.batsmenData; Object.values(batsmen).forEach(b => { window.addBattingRow(); const id = battingRowCount - 1; document.getElementById(`br${id}name`).value = b.batName; document.getElementById(`br${id}runs`).value = b.runs; document.getElementById(`br${id}balls`).value = b.balls; document.getElementById(`br${id}sixes`).value = b.sixes; // Auto-detect Dismissal type for penalties const dis = (b.outDesc || "").toLowerCase(); const sel = document.getElementById(`br${id}dis`); if (dis.includes("not out")) sel.value = "notout"; else if (b.runs === 0 && !dis.includes("not out")) sel.value = "duck"; else sel.value = "out"; }); // --- Bowling Sync (Includes Dots & Maidens) --- const bowlers = innings.bowTeamDetails.bowlersData; Object.values(bowlers).forEach(bw => { window.addBowlingRow(); const id = bowlingRowCount - 1; document.getElementById(`bow${id}name`).value = bw.bowlerName; document.getElementById(`bow${id}overs`).value = bw.overs; document.getElementById(`bow${id}runs`).value = bw.runs; document.getElementById(`bow${id}wkts`).value = bw.wickets; document.getElementById(`bow${id}dots`).value = bw.dots || 0; // Directly from API document.getElementById(`bow${id}maidens`).value = bw.maidens || 0; // For +20 bonus }); }); window.showAlert(`✅ Match ${matchId} Synced Successfully!`, "ok"); document.getElementById('matchFormBody').style.display = 'block'; } catch (err) { window.showAlert(`Sync Failed: ${err.message}`, "err"); console.error("Sync Error Details:", err); } };
